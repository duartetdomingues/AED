void dijkstra(LinkedList **adjList, int V, int finalVertex, FILE *outfp)
{
    LinkedList *aux = NULL;

    int *parent;
    int *weight;
    int i = 0, w = 0;

    parent = (int *)malloc(sizeof(int) * V);
    weight = (int *)malloc(sizeof(int) * V);
    QUEUE *prioQueue = createQueue(V);

    for (i = 0; i < V; i++)
    {
        parent[i] = -1;
        weight[i] = MAX;
    }

    parent[0] = 0;
    weight[0] = 0;
    PQinsert(prioQueue, weight, 0);

    while (prioQueue->free != 0)
    {
        i = PQdelMin(prioQueue, weight);
        if (i == finalVertex)
        {
            fprintf(outfp, "%d\n", weight[i]);
            printPath(adjList, parent, i, outfp);
            fprintf(outfp, "\n");
            free(parent);
            free(weight);
            free(prioQueue->queue);
            free(prioQueue);
            return;
        }

        for (aux = adjList[i]; aux != NULL; aux = aux->next)
        {
            w = aux->Node;
            if (weight[w] == MAX)
            {
                weight[w] = weight[i] + aux->cost;
                parent[w] = i;
                PQinsert(prioQueue, weight, w);
            }

            else if (weight[w] > weight[i] + aux->cost)
            {
                weight[w] = weight[i] + aux->cost;
                parent[w] = i;
                decrease(prioQueue, weight, w);
            }
        }
    }
    fprintf(outfp,"-1\n");
    free(parent);
    free(weight);
    free(prioQueue->queue);
    free(prioQueue);
}

void printPath(WALL **adjList, int *parent, int i, FILE *outfp)
{
    int pai = parent[i];
    static int count = 0;
    if (pai != i)
    {
        count++;
        printPath(adjList, parent, pai, outfp);
    }
    else
    {
        fprintf(outfp, "%d\n", count);
        count = 0;
    }

    WALL *aux = NULL;

    for (aux = adjList[i]; aux != NULL; aux = aux->next)
        if (aux->Node == pai)
        {
            fprintf(outfp, "%d %d %d\n", aux->line + 1, aux->col + 1, aux->cost);
            break;
        }
}

void decrease(QUEUE *prioQ, int *weight, int room)
{
    int i = 0;
    for (i = 0; i < prioQ->free; i++)
    {
        if (prioQ->queue[i] == room)
        {
            FixUp(prioQ, weight, i);
            return;
        }
    }
}

int PQdelMin(QUEUE *prioQ, int *weight)
{ // elimina elemento de maior chave; assume não vazio

    // procura elemento com maior chave
    exch(prioQ->queue[0], prioQ->queue[prioQ->free - 1]);
    FixDown(prioQ, weight, 0, prioQ->free - 1);
    return prioQ->queue[--prioQ->free]; // return à sala
}

void PQinsert(QUEUE *prioQ, int *weight, int room)
{
    if ((prioQ->free + 1) < prioQ->size)
    {
        prioQ->queue[prioQ->free] = room;
        FixUp(prioQ, weight, prioQ->free);
        prioQ->free += 1;
    }
}

void FixUp(QUEUE *prioQ, int *weight, int Idx)
{
    while (Idx > 0 && weight[prioQ->queue[(Idx - 1) / 2]] > weight[prioQ->queue[Idx]])
    {
        exch(prioQ->queue[Idx], prioQ->queue[(Idx - 1) / 2]);
        Idx = (Idx - 1) / 2;
    }
}

void FixDown(QUEUE *prioQ, int *weight, int Idx, int V)
{
    int Child;
    while (2 * Idx < V - 1)
    {
        Child = 2 * Idx + 1;
        if (Child < (V - 1) && weight[prioQ->queue[Child]] > weight[prioQ->queue[Child + 1]])
            Child++;
        if (weight[prioQ->queue[Idx]] < weight[prioQ->queue[Child]])
            break;

        exch(prioQ->queue[Idx], prioQ->queue[Child]);
        Idx = Child;
    }
}
